\paragraph{Global parameters.}
Fix constants
\[
\rho_0 := 1,\qquad C_0 := 1.
\]

\paragraph{IB$_{\rho_0}$.}
A graph $G$ satisfies IB$_{\rho_0}$ if the spectral radius of its nonbacktracking
matrix $B_G$ satisfies
\[
\mathrm{spr}(B_G) \le \rho_0.
\]

\paragraph{CLR$(4,4,2;\rho_0,C_0)$.}
A finite graph $G$ satisfies CLR$(4,4,2;\rho_0,C_0)$ if
\[
\Delta(G)\le 4,\quad G\ \text{is FO$^4$ radius-$2$ homogeneous},\quad
\beta^{\mathrm{loc}}_1(G;2)\le C_0.
\]

\paragraph{Theorem (Cycle--Local Rigidity, fixed parameters).}
For all finite graphs $G$,
\[
\mathrm{IB}_{\rho_0}(G)\ \Longrightarrow\ \mathrm{CLR}(4,4,2;\rho_0,C_0).
\]
